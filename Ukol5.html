const int trigPin = 11;
const int echoPin = 12;
const int LED_ZELENA = 2;
const int LED_ZLUTA = 3;
const int LED_CERVENA = 4;

unsigned long t_ms;
long duration;
int distance;
bool auto_je_u_semaforu = false;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(LED_ZELENA, OUTPUT);
  pinMode(LED_ZLUTA, OUTPUT);
  pinMode(LED_CERVENA, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  t_ms = millis();

  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;

  if (distance > 0 && distance < 15) {
    if (auto_je_u_semaforu == false) {
      digitalWrite(LED_ZELENA, LOW);
      digitalWrite(LED_ZLUTA, HIGH);
      digitalWrite(LED_CERVENA, LOW);
      delay(2000);

      digitalWrite(LED_ZLUTA, LOW);
      digitalWrite(LED_CERVENA, HIGH);
      delay(5000);

      digitalWrite(LED_ZLUTA, HIGH);
      delay(1500);

      digitalWrite(LED_CERVENA, LOW);
      digitalWrite(LED_ZLUTA, LOW);
      digitalWrite(LED_ZELENA, HIGH);
      
      auto_je_u_semaforu = true;
    }
  } 
  else {
    auto_je_u_semaforu = false;
    
    digitalWrite(LED_ZELENA, LOW);
    digitalWrite(LED_CERVENA, LOW);

    if (t_ms % 1000 < 500) {
      digitalWrite(LED_ZLUTA, HIGH);
    } else {
      digitalWrite(LED_ZLUTA, LOW);
    }
  }
}
